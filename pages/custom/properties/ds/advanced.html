Make Real Custom Properties


:root {
  // For use later with the Paint API
  @property --theme-color {
    syntax: 'blue|green|red|yellow|grey';
    inherits: true;
    initial-value: blue;
  }

  // For use later with the Layout API
  @property --columns {
    syntax: '<integer> | auto';
    inherits: false;
    initial-value: 2;
  }

  @property --padding {
    syntax: '<integer>';
    inherits: true;
    initial-value: 0;
  }
}
