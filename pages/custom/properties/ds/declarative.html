Register in CSS


Proposed CSS Syntax

@property --my-color {
  syntax: '<color>';
}

JS Equivalent

if ('registerProperty' in CSS) {
  window.CSS.registerProperty({
    name: '--my-color',
    syntax: '<color>',
  });
}
